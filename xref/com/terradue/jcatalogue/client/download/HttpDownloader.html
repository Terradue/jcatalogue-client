27  @Data
28  @Protocol({ "http", "https" })
29  public final class HttpDownloader

31  {
32  
33      private static final String RW = "rw";

37      @Getter( AccessLevel.NONE )
38      private final AsyncHttpClient httpClient;

40      private final Logger logger = LoggerFactory.getLogger( getClass() );
41  
42      private final Map<String, Realm> realms = new HashMap<String, Realm>();
43  
44      public void registerRealm( String host, Realm realm )
45      {
46          if ( host == null )
47          {
48              throw new IllegalArgumentException( "Input host must be not null" );
49          }
50          if ( realm == null )
51          {
52              throw new IllegalArgumentException( "Input realm must be not null" );
53          }
54  
55          realms.put( host, realm );
56      }
57  
58      @Override
59      public void download( File targetDir, URI fileUri, DownloadHandler handler )
60      {
61          String fileName = fileUri.getPath().substring( fileUri.getPath().lastIndexOf( '/' ) + 1 );
62  
63          File targetFile = new File( targetDir, fileName );
64  
65          if ( logger.isInfoEnabled() )
66          {
67              logger.info( "Downloading {} to {}...", fileUri, fileName );
68          }
69  
70          ResumableAsyncHandler<Response> resumableHandler = new ResumableAsyncHandler<Response>();
71          try
72          {
73              resumableHandler.setResumableListener( new ResumableRandomAccessFileListener( new RandomAccessFile( targetFile,
74                                                                                                                  RW ) ) );
75          }
76          catch ( FileNotFoundException e )
77          {
78              // TODO verify, but that should not happen
79          }
80  
81          RequestBuilder requestBuilder = new RequestBuilder( GET ).setUrl( fileUri.toString() );
82  
83          if ( realms.containsKey( fileUri.getHost() ) )
84          {
85              requestBuilder.setRealm( realms.get( fileUri.getHost() ) );
86          }
87  
88          Response response;
89          try
90          {
91              response = httpClient.executeRequest( requestBuilder.build(), resumableHandler ).get();
92  
93              if ( HTTP_OK != response.getStatusCode() )
94              {
95                  handler.onError( format( "Impossible to download %s from %s, server replied %s",
96                                           targetFile.getName(), fileUri, response.getStatusText() ) ) ;
97                  targetFile.delete();
98              }
99              else
100             {
101                 handler.onSuccess( targetFile );
102             }
103         }
104         catch ( Exception e )
105         {
106             handler.onError( e );
107         }
108     }
