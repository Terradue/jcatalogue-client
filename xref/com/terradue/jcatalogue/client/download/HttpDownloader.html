24  @Data
25  @Protocol({ "http", "https" })
26  public final class HttpDownloader
27      implements Downloader
28  {
29  
30      private static final String RW = "rw";
31  
32      private static final String GET = "GET";
33  
34      @Getter( AccessLevel.NONE )
35      private final AsyncHttpClient httpClient;
36  
37      private final Map<String, Realm> realms = new HashMap<String, Realm>();
38  
39      public void registerRealm( String host, Realm realm )
40      {
41          if ( host == null )
42          {
43              throw new IllegalArgumentException( "Input host must be not null" );
44          }
45          if ( realm == null )
46          {
47              throw new IllegalArgumentException( "Input realm must be not null" );
48          }
49  
50          realms.put( host, realm );
51      }
52  
53      @Override
54      public void download( File targetDir, URI fileUri, DownloadHandler handler )
55      {
56          String fileName = fileUri.getPath().substring( fileUri.getPath().lastIndexOf( '/' ) + 1 );
57  
58          File targetFile = new File( targetDir, fileName );
59  
60          ResumableAsyncHandler<Response> resumableHandler = new ResumableAsyncHandler<Response>();
61          try
62          {
63              resumableHandler.setResumableListener( new ResumableRandomAccessFileListener( new RandomAccessFile( targetFile,
64                                                                                                                  RW ) ) );
65          }
66          catch ( FileNotFoundException e )
67          {
68              // TODO verify, but that should not happen
69          }
70  
71          RequestBuilder requestBuilder = new RequestBuilder( GET ).setUrl( fileUri.toString() );
72  
73          if ( realms.containsKey( fileUri.getHost() ) )
74          {
75              requestBuilder.setRealm( realms.get( fileUri.getHost() ) );
76          }
77  
78          Response response;
79          try
80          {
81              response = httpClient.executeRequest( requestBuilder.build(), resumableHandler ).get();
82  
83              if ( HTTP_OK != response.getStatusCode() )
84              {
85                  handler.onError( format( "Impossible to download file %s, server replied %s",
86                                           fileUri, response.getStatusText() ) ) ;
87              }
88  
89              handler.onSuccess( targetFile );
90          }
91          catch ( Exception e )
92          {
93              handler.onError( e );
94          }
95      }
