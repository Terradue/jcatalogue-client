77      public <T> T invoke( HttpMethod httpMethod, URI uri, AsyncHandler<T> handler, Parameter...parameters )
78      {
79          checkNotNull( uri, "Input URI cannot be null" );
80  
81          RequestBuilder requestBuilder = new RequestBuilder( httpMethod.toString() ).setUrl( uri.toString() );
82  
83          // send all domain cookies, if any
84          Map<String, Cookie> domainCookies = cookiesRegistry.get( uri.getHost() );
85          if ( domainCookies != null && !domainCookies.isEmpty() )
86          {
87              for ( Cookie cookie : domainCookies.values() )
88              {
89                  requestBuilder.addCookie( cookie );
90              }
91          }
92  
93          // send all parameters
94          for ( Parameter parameter : parameters )
95          {
96              if ( HttpMethod.GET == httpMethod )
97              {
98                  requestBuilder.addQueryParameter( parameter.getName(), parameter.getValue() );
99              }
100             else
101             {
102                 requestBuilder.addParameter( parameter.getName(), parameter.getValue() );
103             }
104         }
105 
106         // send the realms authentication, if any
107         if ( realms.containsKey( uri.getHost() ) )
108         {
109             requestBuilder.setRealm( realms.get( uri.getHost() ) );
110         }
111 
112         try
113         {
114             return httpClient.executeRequest( requestBuilder.build(), handler ).get();
115         }
116         catch ( Exception e )
117         {
118             throw new RuntimeException( "An error occurred while invoking " + uri, e );
119         }
120     }

145     public void registerSSLCerificates( File sslCertificate, File sslKey, String sslPassword )
146     {
147         checkFile( sslCertificate );
148         checkFile( sslKey );
149 
150         if ( sslPassword == null )
151         {
152             sslPassword = "";
153         }
154 
155         final char[] password = sslPassword.toCharArray();
156 
157         try
158         {
159             final KeyStore store = new KeyMaterial( sslCertificate, sslKey, password ).getKeyStore();
160             store.load( null, password );
161 
162             // initialize key and trust managers -> default behavior
163             final KeyManagerFactory keyManagerFactory = KeyManagerFactory.getInstance( "SunX509" );
164             // password for key and store have to be the same IIRC
165             keyManagerFactory.init( store, password );
166 
167             keyManagers.addAll( asList( keyManagerFactory.getKeyManagers() ) );
168 
169             httpClient.getConfig().getSSLContext().init( keyManagers.toArray( new KeyManager[keyManagers.size()] ),
170                                                          trustManagers,
171                                                          null );
172         }
173         catch ( Exception e )
174         {
175             throw new IllegalStateException( "Impossible to initialize SSL certificate/key", e );
176         }
177     }
